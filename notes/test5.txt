package main

import (
	"fmt"
	"sync"
	"time"
)

func source(id int, out chan<- string, wg *sync.WaitGroup) {
	defer wg.Done()
	delay := time.Duration((id+1)*100) * time.Millisecond
	time.Sleep(delay)
	out <- fmt.Sprintf("Data dari sumber %d", id)
}

func processor(in1 <-chan string, in2 <-chan string, out chan<- string, wg *sync.WaitGroup, mu *sync.Mutex) {
	defer wg.Done()
	for {
		select {
		case msg := <-in1:
			mu.Lock()
			processedMsg := fmt.Sprintf("Diproses dari sumber 1: %s", msg)
			mu.Unlock()
			out <- processedMsg
		case msg := <-in2:
			mu.Lock()
			processedMsg := fmt.Sprintf("Diproses dari sumber 2: %s", msg)
			mu.Unlock()
			out <- processedMsg
		}
	}
}

func main() {
	var wg sync.WaitGroup
	var mu sync.Mutex

	chan1 := make(chan string)
	chan2 := make(chan string)
	resultChan := make(chan string)
	done := make(chan bool)

	// Tambahkan counter untuk goroutine sumber
	wg.Add(2)
	go source(1, chan1, &wg)
	go source(2, chan2, &wg)

	// Tambahkan counter untuk goroutine processor
	wg.Add(1)
	go processor(chan1, chan2, resultChan, &wg, &mu)

	// Baca hasil sampai kedua sumber selesai dan channel ditutup
	go func() {
		wg.Wait()
		close(resultChan)
		close(done)
	}()

	for result := range resultChan {
		fmt.Println("Hasil:", result)
	}

	<-done
	fmt.Println("Program selesai")
}
